FCM

J(U, C)=\sum_{s=1}^{N} \sum_{i=1}^{K}\left(u_{s i}\right)^{m} * distance(x_{s},c_{i}) \nonumber s.t. \nonumber \sum_{i=1}^{K} u_{si}=1,s=1,2,···，N-1,N

​ 由拉格朗日常系数法可得：

F(U,C,\Lambda)=\sum_{s=1}^{N} (\sum_{i=1}^{K}\left(u_{s i}\right)^{m} * distance(x_{s},c_{i})-\lambda_s( \sum_{i=1}^{K}(u_{si})-1)) \nonumber

\begin{align} \frac {\partial F}{\partial u_{si}}=m u_{si}^{m-1}*distance(x_s,c_i)-\lambda_s =0 \end{align}

由(1)(1)(1)式可得：

\begin{align} u_{si}=(\frac {\lambda_s}{m})^{\frac {1}{m-1}}*(\frac {1}{distance(x_s,c_i)})^{\frac {1}{m-1}} \end{align}

\begin{align} \frac {\partial F}{\partial \lambda_{s}}= \sum_{i=1}^{K}(u_{si})-1=0 \end{align}

把(2)(2)(2)式带入(3)(3)(3)式可得：

\sum_{i=1}^{K}((\frac {\lambda_s}{m})^{\frac {1}{m-1}}*(\frac {1}{distance(x_s,c_i)})^{\frac {1}{m-1}})-1=0 \nonumber \\ (\frac {\lambda_s}{m})^{\frac {1}{m-1}} * \sum_{i=1}^{K}(\frac {1}{distance(x_s,c_i)})^{\frac {1}{m-1}}-1=0 \\

(λsm)1m−1=1∑i=1K(1distance(xs,ci))1m−1(\frac {\lambda_s}{m})^{\frac {1}{m-1}}=\frac{1}{\sum_{i=1}^{K}(\frac {1}{distance(x_s,c_i)})^{\frac {1}{m-1}}} (mλs​​)m−11​=∑i=1K​(distance(xs​,ci​)1​)m−11​1​

把(4)(4)(4)式带入(2)(2)(2)式可得：

\begin{align} u_{si}&=\frac{1}{\sum_{j=1}^{K}(\frac {1}{distance(x_s,c_j)})^{\frac {1}{m-1}}}*(\frac {1}{distance(x_s,c_i)})^{\frac {1}{m-1}} \\ &=\frac{1}{distance(x_s,c_i) \sum_{j=1}^{K}(\frac {1}{distance(x_s,c_j)})} \end{align}

\begin{align} \frac {\partial F}{\partial c_{i}}&=\frac {\partial \left (\sum_{s=1}^{N} \sum_{i=1}^{K}u_{s i}^{m} * distance(x_{s},c_{i}) \right)}{\partial c_{i}} \\ &=\frac {\partial \left (\sum_{i=1}^{K} \sum_{s=1}^{N}u_{s i}^{m} * distance(x_{s},c_{i}) \right)}{\partial c_{i}} \\ &=\sum_{s=1}^{N}u_{s i}^{m} * \left (\frac{\partial distance(x_s,c_i)}{\partial c_i} \right) \end{align}

结论：

\begin{align} u_{si}&=\frac{1}{distance(x_s,c_i) \sum_{j=1}^{K}(\frac {1}{distance(x_s,c_j)})} \\ \frac {\partial F}{\partial c_{i}} &=\sum_{s=1}^{N}u_{s i}^{m} * \left (\frac{\partial distance(x_s,c_i)}{\partial c_i} \right)=0 \end{align}

1-cos(θ\thetaθ)

\vec{d_1}=\vec{x_s}-\bar{x_s} \\ \vec{d_2}=\vec{c_i}-\bar{c_i}\\ \begin{align} distance(x_s,c_i)&=1-\frac{\vec{d_1} *\vec{d_2}}{||d_1||*||d_2||}\\ &=1-\frac{d_1^T *d_2}{||d_1||*||d_2||}\\ \end{align}

\begin{align} \frac{\partial distance(x_s,c_i)}{\partial c_i}&=\frac{\partial distance(x_s,c_i)}{\partial (c_i-\bar c_i)}*\frac{\partial (c_i-\bar c_i)}{\partial c_i}=\frac{\partial distance(x_s,c_i)}{\partial d_2}*\frac{\partial (c_i-\bar c_i)}{\partial c_i}\\ &=\frac{\partial distance(x_s,c_i)}{\partial d_2}*(I-\frac{1}{N} I)\\ &=-\frac{\partial F}{\partial d_2}*(I-\frac{1}{N} I) \end{align}

F=d1Td2∣∣d1∣∣∗∣∣d2∣∣F=\frac{d_1^Td_2}{||d_1||*||d_2||} F=∣∣d1​∣∣∗∣∣d2​∣∣d1T​d2​​

∂∣∣x∣∣∂x=∂(xTx)∂xtx∗∂xTx∂x=12xtx∗2x=xxtx=x∣∣x∣∣\frac{\partial ||x||}{\partial x}=\frac{\partial \sqrt{(x^T x)}}{\partial x^t x}*\frac{\partial x^T x}{\partial x}=\frac{1}{2\sqrt{x^t x}}*2x=\frac{x}{\sqrt{x^t x}}=\frac{x}{||x||} ∂x∂∣∣x∣∣​=∂xtx∂(xTx)​​∗∂x∂xTx​=2xtx​1​∗2x=xtx​x​=∣∣x∣∣x​

\begin{align} \frac{\partial F}{\partial d_2}&=\frac{d_1*||d_2||-\frac{d_1^T d_2^2}{||d_2||}}{||d_1||*||d_2||^2}\\ &=\frac{d_1*||d_2||^2-{d_1^T d_2^2}}{||d_1||*||d_2||^3} \end{align}

\begin{align} \frac{\partial distance(x_s,c_i)}{\partial c_i}&=-\frac{d_1*||d_2||^2-{d_1^T d_2^2}}{||d_1||*||d_2||^3}*(I-\frac{1}{N} I) \end{align}

\begin{align} \frac {\partial F}{\partial c_{i}} &=\sum_{s=1}^{N}u_{s i}^{m} * \left (\frac{\partial distance(x_s,c_i)}{\partial c_i} \right)=0\\ &\Rightarrow \sum_{s=1}^{N}u_{s i}^{m}*-\frac{d_1*||d_2||^2-{d_1^T d_2^2}}{||d_1||*||d_2||^3}*(I-\frac{1}{N} I)=0\\ &\Rightarrow \sum_{s=1}^{N}u_{s i}^{m}*(d_1*||d_2||^2-d_1^T d_2^2)=0\\ &\Rightarrow ||d_2||^2*\sum_{s=1}^{N}u_{s i}^{m}d_1=\sum_{s=1}^{N}u_{s i}^{m}d_1^T d_2 d_2\\ &\Rightarrow ||d_2||^2*\sum_{s=1}^{N}u_{s i}^{m}d_1=(\sum_{s=1}^{N}u_{s i}^{m}d_1^T d_2)* d_2\\ &\Rightarrow d_2=\frac{||d_2||^2*\sum_{s=1}^{N}u_{s i}^{m}d_1}{\sum_{s=1}^{N}u_{s i}^{m}d_1^T d_2}\\ &\Rightarrow d_2=k*\sum_{s=1}^{N}u_{s i}^{m}d_1,k=\frac{||d_2||^2}{\sum_{s=1}^{N}u_{s i}^{m}d_1^T d_2}\\ &\Rightarrow c_i=\bar{c_i}+k*\sum_{s=1}^{N}u_{s i}^{m}d_1,k=\frac{||d_2||^2}{\sum_{s=1}^{N}u_{s i}^{m}d_1^T d_2} \end{align}

结论

综上所述，当distance(xs,ci)=1−(xs−xˉs)T∗(ci−ciˉ)∣∣xs−xˉs∣∣∗∣∣ci−ciˉ∣∣distance(x_s,c_i)=1-\frac{( x_s-\bar x_s )^T *(c_i-\bar{c_i})}{|| x_s-\bar x_s||*||c_i-\bar{c_i}||}distance(xs​,ci​)=1−∣∣xs​−xˉs​∣∣∗∣∣ci​−ci​ˉ​∣∣(xs​−xˉs​)T∗(ci​−ci​ˉ​)​时

\begin{align} u_{si}&=\frac{1}{distance(x_s,c_i) \sum_{j=1}^{K}(\frac {1}{distance(x_s,c_j)})} \\ c_i&=\bar{c_i}+k*\sum_{s=1}^{N}u_{s i}^{m}d_1,k=\frac{||d_2||^2}{\sum_{s=1}^{N}u_{s i}^{m}d_1^T d_2} \end{align}

arccos(θ\thetaθ)

\vec{d_1}=\vec{x_s}-\bar{x_s} \\ \vec{d_2}=\vec{c_i}-\bar{c_i}\\ \begin{align} distance(x_s,c_i)&=acrcos(\frac{\vec{d_1} *\vec{d_2}}{||d_1||*||d_2||})\\ &=acrcos(\frac{d_1^T *d_2}{||d_1||*||d_2||})\\ \end{align}

\begin{align} \frac{\partial distance(x_s,c_i)}{\partial c_i}&=\frac{\partial distance(x_s,c_i)}{\partial (c_i-\bar c_i)}*\frac{\partial (c_i-\bar c_i)}{\partial c_i}=\frac{\partial distance(x_s,c_i)}{\partial d_2}*\frac{\partial (c_i-\bar c_i)}{\partial c_i}\\ &=-\frac{\frac{\partial F}{\partial d_2}}{\sqrt{1-F^2}}*(I-\frac{1}{N} I) \end{align}

F=d1Td2∣∣d1∣∣∗∣∣d2∣∣F=\frac{d_1^Td_2}{||d_1||*||d_2||} F=∣∣d1​∣∣∗∣∣d2​∣∣d1T​d2​​

∂∣∣x∣∣∂x=∂(xTx)∂xtx∗∂xTx∂x=12xtx∗2x=xxtx=x∣∣x∣∣\frac{\partial ||x||}{\partial x}=\frac{\partial \sqrt{(x^T x)}}{\partial x^t x}*\frac{\partial x^T x}{\partial x}=\frac{1}{2\sqrt{x^t x}}*2x=\frac{x}{\sqrt{x^t x}}=\frac{x}{||x||} ∂x∂∣∣x∣∣​=∂xtx∂(xTx)​​∗∂x∂xTx​=2xtx​1​∗2x=xtx​x​=∣∣x∣∣x​

\begin{align} \frac{\partial F}{\partial d_2}&=\frac{d_1*||d_2||-\frac{d_1^T d_2^2}{||d_2||}}{||d_1||*||d_2||^2}\\ &=\frac{d_1*||d_2||^2-{d_1^T d_2^2}}{||d_1||*||d_2||^3} \end{align}

\begin{align} \frac{\partial distance(x_s,c_i)}{\partial c_i}&=-\frac {\frac{d_1*||d_2||^2-{d_1^T d_2^2}}{||d_1||*||d_2||^3}}{\sqrt{1-(\frac{d_1^Td_2}{||d_1||*||d_2||})^2} }*(I-\frac{1}{N} I)\\ &=-\frac {d_1*||d_2||^2-d_1^T d_2^2}{||d_2||^2\sqrt{||d_1||^2||d_2||^2-(d_1^Td_2)^2 } }*(I-\frac{1}{N} I)\\ &=-\frac {N-1}{N||d_2||^2\sqrt{||d_1||^2||d_2||^2-(d_1^Td_2)^2 } }*(||d_2||^2*d_1-d_1^T d_2*d_2) \end{align}

\begin{align} \frac {\partial F}{\partial c_{i}} &=\sum_{s=1}^{N}u_{s i}^{m} * \left (\frac{\partial distance(x_s,c_i)}{\partial c_i} \right)=0\\ &\Rightarrow \sum_{s=1}^{N}u_{s i}^{m} \frac {N-1}{N||d_2||^2\sqrt{||d_1||^2||d_2||^2-(d_1^Td_2)^2 } }*(||d_2||^2*d_1-d_1^T d_2*d_2)=0\\ &\Rightarrow \sum_{s=1}^{N}u_{s i}^{m} \frac {N-1}{N||c_i-\bar {c_i}||^2\sqrt{||x_s-\bar {x_s}||^2||c_i-\bar {c_i}||^2-((x_s-\bar {x_s}^T) (c_i-\bar {c_i}))^2 } }*(||c_i-\bar {c_i}||^2* \nonumber\\ &(x_s-\bar {x_s})-(x_s-\bar {x_s}^T) (c_i-\bar {c_i})*(c_i-\bar {c_i}))=0 \nonumber\\ \end{align}

\begin{align} &\Rightarrow \sum_{s=1}^{N}u_{s i}^{m} \frac {N-1}{N||d_2||^2\sqrt{||d_1||^2||d_2||^2-(d_1^Td_2)^2 } }*(||d_2||^2*d_1-d_1^T d_2*d_2)=0\\ &\Rightarrow \sum_{s=1}^{N}u_{s i}^{m} \frac {1}{\sqrt{||d_1||^2||d_2||^2-(d_1^Td_2)^2 } }*(||d_2||^2*d_1-d_1^T d_2*d_2)=0 \\ &\Rightarrow \sum_{s=1}^{N}u_{s i}^{m} \frac {1}{\sqrt{||d_1||^2||d_2||^2-(d_1^Td_2)^2 } }*(d_2^T d_2 *d_1-d_2^T d_1*d_2)=0 \\ &\Rightarrow d_2^T\sum_{s=1}^{N} \frac {u_{s i}^{m} *d2*d_1}{\sqrt{||d_1||^2||d_2||^2-(d_1^Td_2)^2 } } = d_2^T\sum_{s=1}^{N} \frac {u_{s i}^{m}*d_1*d_2}{\sqrt{||d_1||^2||d_2||^2-(d_1^Td_2)^2 } }\\ &\Rightarrow d_2^T*d_2\sum_{s=1}^{N} \frac {u_{s i}^{m} *d_1}{\sqrt{||d_1||^2||d_2||^2-(d_1^Td_2)^2 } } = d_2^T*\sum_{s=1}^{N} \frac {u_{s i}^{m}*d_1}{\sqrt{||d_1||^2||d_2||^2-(d_1^Td_2)^2 } } *d_2\\ &\Rightarrow d_2=\frac{ d_2^T*d_2\sum_{s=1}^{N} \frac {u_{s i}^{m} *d_1}{\sqrt{||d_1||^2||d_2||^2-(d_1^Td_2)^2 } } }{ d_2^T*\sum_{s=1}^{N} \frac {u_{s i}^{m}*d_1}{\sqrt{||d_1||^2||d_2||^2-(d_1^Td_2)^2 } } } \end{align}

结论

综上所述，当距离函数为distance(xs,ci)=arccos((xs−xˉs)T∗(ci−ciˉ)∣∣xs−xˉs∣∣∗∣∣ci−ciˉ∣∣)distance(x_s,c_i)=arccos(\frac{( x_s-\bar x_s )^T *(c_i-\bar{c_i})}{|| x_s-\bar x_s||*||c_i-\bar{c_i}||})distance(xs​,ci​)=arccos(∣∣xs​−xˉs​∣∣∗∣∣ci​−ci​ˉ​∣∣(xs​−xˉs​)T∗(ci​−ci​ˉ​)​)时：

\begin{align} u_{si}&=\frac{1}{distance(x_s,c_i) \sum_{j=1}^{K}(\frac {1}{distance(x_s,c_j)})} \\ c_i&=\bar{c_i}+\frac{ d_2^T*d_2\sum_{s=1}^{N} \frac {u_{s i}^{m} *d_1}{\sqrt{||d_1||^2||d_2||^2-(d_1^Td_2)^2 } } }{ d_2^T*\sum_{s=1}^{N} \frac {u_{s i}^{m}*d_1}{\sqrt{||d_1||^2||d_2||^2-(d_1^Td_2)^2 } } },d_1={x_s}-\bar{x_s} , d_2=c_i-\bar{c_i}\\ \end{align}
